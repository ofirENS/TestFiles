258function vals = BoneStar(k,spacePoints,t,timePoints)
259% Calculate the matrix Aone for a specific space point x_k, for all times
260% t_i in A
261% points x_k=1:numel(spacePoints)
262vals = zeros(numel(t),2*numel(t));
263for j = 1:numel(t)
264    for i = 1:numel(t)
265        b1ki = Bcoeff(1,j,spacePoints(k),t(i),timePoints);
266        vals(j,2*i-1:2*i) =[b1ki,...
267                           -b1ki];
268    end
269end
270
271end
