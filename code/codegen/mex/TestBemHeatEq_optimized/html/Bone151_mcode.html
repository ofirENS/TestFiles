245function vals = Bone(k,spacePoints,t,timePoints)
246% Calculate the matrix Aone for a specific space point x_k, for all times
247% t_i in A
248% points x_k=1:numel(spacePoints)
249vals = zeros(numel(t),2*numel(t));
250for j = 1:numel(t)
251    for i = 1:numel(t)
252        vals(j,2*i-1:2*i) =[Bcoeff(0,j,spacePoints(k),t(i),timePoints),...
253                            Bcoeff(1,j,spacePoints(k),t(i),timePoints)];
254    end
255end
256end
