1function [u,r]= TestBemHeatEq_optimized(sigIn,regOrder,lambda,alpha,numSpacePoints,plotFlag)
  2% Solve the inveerse heat equation with an unknown source
  3% u_t(x,t) = ku_{xx}(x,y) +r(t)*f(x,t)
  4% with 0<=x<=1, 0<=t<=T
  5% subject to the boundary conditions u(0,t) = u(1,t)
  6% and u_x(0,t)+alpha*u(0,t)=0
  7% and initial condition
  8% u(x,0) = phi(x) =u0, 0<=x<=1
  9% In this problem we take f(x,t) =1 for all x and t
 10% lambda    regularization constant (affects smoothness of source solution)
 11% regOrder  Tikhonov regularization term order, [0,1,or 2]
 12
 13
 14% The convention for most matrices is that rows are time, columns are space
 15% alpha    = -1e-14; % for the boundary condition
 16
 17if mod(numSpacePoints+1,2)~=0
 18    numSpacePoints=numSpacePoints+1;% make it odd
 19end
 20N0 = numSpacePoints; % space points;
 21E  = (sigIn)/2;      % int(u(x,t)dx) is a time function of size 1xN % assuming the temp drops to 0 at the boundaries linearly
 22N  = numel(sigIn);   % time points
 23f  = zeros(N,N0);    % function of N time points over N0 space points
 24f(:,ceil(N0/2)) = 1; % the source is only active at the mid point x=0.5;
 25
 26% determine initial wave 
 27u0       = zeros(1,numSpacePoints);
 28midPoint = ceil(numSpacePoints/2);
 29u0m      = max(sigIn(1:min(10,numel(sigIn)))); % take the max value 
 30% u0       = u0m*ones(1,N0);
 31% linearly decrease to zero to match with boundary conditions
 32u0(midPoint)       = u0m;
 33u0(1:midPoint-1)   = ((0:(midPoint-2))./midPoint)*u0m;
 34u0(midPoint+1:end) = (((midPoint-2):-1:0)./midPoint)*u0m;
 35 % is the Phi function
 36
 37% Determine the boundary and initial functions
 38% du/dn(0,t) = q0j
 39q0j = zeros(N,1);
 40%du/dn(1,t) = q1j
 41q1j = zeros(N,1);
 42% u(0,t) = h0j
 43h0j = zeros(N,1);
 44% u(1,t) = h1j
 45h1j = zeros(N,1);
 46
 47
 48% Discretize the time and space domain
 49T      = numel(sigIn);         % max time;
 50gridT  = (0:(N)).*(T/N);% N+1 points
 51t      = zeros(1,N);
 52for tIdx = 1:N
 53    t(tIdx) = (gridT(tIdx)+gridT(tIdx+1))/2;
 54end
 55gridX = (0:(N0))./N0;
 56x     = zeros(1,N0);
 57for xIdx = 1:numel(x)
 58    x(xIdx) = (gridX(xIdx+1)+gridX(xIdx))/2;
 59end
 60
 61%===========================
 62
 63% Calculate the Boundary Elements
 64
 65% Calculate A and B, B*, and D
 66A      = zeros(2*N,2*N);
 67B      = zeros(2*N,2*N);
 68Bstar  = zeros(2*N,2*N);
 69D      = zeros(2*N,N);
 70for j = 1:N % time
 71    for i = 1:N % time
 72        deltaij = double(i==j);
 73
 74        A((2*i-1):2*i,(2*j-1):2*j) = [Acoeff(0,j,0,t(i),gridT), Acoeff(1,j,0,t(i),gridT);...
 75                                      Acoeff(0,j,1,t(i),gridT), Acoeff(1,j,1,t(i),gridT)];
 76        % temp terms
 77        b1j0i = Bcoeff(1,j,0,t(i),gridT);
 78        b1j1i = Bcoeff(1,j,1,t(i),gridT);
 79        
 80        B((2*i-1):2*i,(2*j-1):2*j) = [Bcoeff(0,j,0,t(i),gridT)+0.5*deltaij, b1j0i;...
 81                                      Bcoeff(0,j,1,t(i),gridT), b1j1i+0.5*deltaij];
 82        
 83        Bstar((2*i-1):2*i,(2*j-1):2*j) = [b1j0i,            -b1j0i;...
 84                                          b1j1i+deltaij/2, -b1j1i-0.5*deltaij];
 85        
 86        D((2*i-1):2*i,j)=[Dcoeff(x,j,0,t(i),gridT,f(i,:));...
 87                        Dcoeff(x,j,1,t(i),gridT,f(i,:))];
 88    end
 89end
 90
 91% Calculate C
 92C = zeros(2*N,N0);
 93for k = 1:N0% space
 94    for i=1:N% time
 95        C((2*i-1):2*i,k) = [Ccoeff(k,0,t(i),gridX);
 96                            Ccoeff(k,1,t(i),gridX)];
 97    end
 98end
 99
100% Solve the equation
101% r =((X'*X+lambda*regTerm)^-1)*X'Y, with r the source and X and Y are to be
102% calculated
103
104% Calculate the components of X and Y
105s1 = zeros(N,N,N0);
106s2 = zeros(N,N0);
107for k = 1:N0% space variable
108    % components of X
109%     a1k = Aone(k,x,t,gridT); % replaced
110%     b1k = Bone(k,x,t,gridT);
111%     b1s = BoneStar(k,x,t,gridT); 
112    [a1k,b1k,b1s] = CalculateA1B1B1Star(k,x,t,gridT);
113    ax  = (a1k-(1/alpha)*(b1k+b1s))/... %(a1k-(1/alpha)*(b1k-b1s))
114          (A-(1/alpha)*(B+Bstar));
115    s1(:,:,k) = (-ax*D + Done(k,x,t,gridT,f));
116    
117    % components of Y
118    ay = (a1k-(1/alpha)*(b1k+b1s))/...
119         (A-(1/alpha)*(B+Bstar));
120    s2(:,k) = (ay*C-Cone(x,gridX,t))*u0' ;
121end
122
123X = (1/N0)*sum(s1,3);
124
125Y = E'+(1/N0)*sum(s2,2);
126
127regTerm = GetRegularizationTerm(N,regOrder); % get RregTerm= 'R term
128% The source solution with regularization coefficient lambda is
129r   = ((X')*X+lambda*regTerm)\(X'*Y);
130
131% Calculate the solution u(x,t)
132eta     = ones(1,N0);
133eta(1)  = 0.5;
134eta(N0) = 0.5;
135u       = zeros(N,N0);
136for j=1:N0    % space index
137    for i=1:N % time index
138        sTime   = 0;% cumulative sums
139        sSpace  = 0;
140        
141        for tIdx = 1:N % time index
142            sTime = sTime+ Acoeff(0,tIdx,x(j),t(i),gridT)*q0j(tIdx) + Acoeff(1,tIdx,x(j),t(i),gridT)*q1j(tIdx)-...
143                Bcoeff(0,tIdx,x(j),t(i),gridT)*h0j(tIdx) - Bcoeff(1,tIdx,x(j),t(i), gridT)*h1j(tIdx)+...
144                Dcoeff(x,tIdx,x(j),t(i),gridT,f(i,:))*r(tIdx);
145        end
146        
147        for ssIdx = 1:N0 % space index
148            sSpace = sSpace +Ccoeff(ssIdx,x(j),t(i),gridT)*u0(ssIdx);
149        end
150        u(i,j) = (sTime+sSpace)*eta(j);
151    end
152end
153
154% plot source, signal and solution
155if all(plotFlag)
156  PlotResults(t,sigIn,u,r);
157end
158end
