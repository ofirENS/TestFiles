1function [u,r]= TestBemHeatEq_optimized(sigIn,regOrder,lambda,alpha,numSpacePoints,initCond,plotFlag)
  2% Solve the inveerse heat equation with an unknown source
  3% u_t(x,t) = ku_{xx}(x,y) +r(t)*f(x,t)
  4% with 0<=x<=1, 0<=t<=T
  5% subject to the boundary conditions u(0,t) = u(1,t)
  6% and u_x(0,t)+alpha*u(0,t)=0
  7% and initial condition
  8% u(x,0) = phi(x) =u0, 0<=x<=1
  9% In this problem we take f(x,t) =1 for all x and t
 10% lambda    regularization constant (affects smoothness of source solution)
 11% regOrder  Tikhonov regularization term order, [0,1,or 2]
 12% initCond= 1 or 2, for initCond=1, the h0 and h1 function are sigIn
 13% for initCond = 2, the h0=h1=0, and the energy drops linearly to 0
 14
 15% The convention for most matrices is that rows are time, columns are space
 16% alpha    = -1e-14; % for the boundary condition
 17
 18if mod(numSpacePoints+1,2)~=0
 19    numSpacePoints=numSpacePoints+1;% make it odd
 20end
 21N0 = numSpacePoints; % space points;
 22N  = numel(sigIn);   % time points
 23f  = zeros(N,N0);    % function of N time points over N0 space points
 24f(:,ceil(N0/2)) = 1; % the source is only active at the mid point x=0.5;
 25
 26if initCond==1
 27E  = (sigIn)*N0;      % int(u(x,t)dx) is a time function of size 1xN % assuming the temp drops to 0 at the boundaries linearly
 28
 29% determine initial wave u(x,0) is the Phi function
 30
 31u0m      = max(sigIn(1:min(10,numel(sigIn)))); % take the max value 
 32u0       = u0m*ones(1,N0); % u(x,0)
 33
 34% Determine the boundary and initial functions
 35% du/dn(0,t) = q0j
 36q0j = zeros(N,1); % the flux 
 37%du/dn(1,t) = q1j
 38q1j = zeros(N,1); % the flux 
 39% u(0,t) = h0j
 40h0j = sigIn;%zeros(N,1);
 41% u(1,t) = h1j
 42h1j = sigIn;% zeros(N,1);
 43
 44elseif initCond==2
 45    
 46   E  = (sigIn)/2;      % int(u(x,t)dx) is a time function of size 1xN % assuming the temp drops to 0 at the boundaries linearly
 47% determine initial wave u(x,0) is the Phi function
 48u0       = zeros(1,numSpacePoints);
 49midPoint = ceil(numSpacePoints/2);
 50u0m      = max(sigIn(1:min(10,numel(sigIn)))); % take the max value 
 51% u0       = u0m*ones(1,N0); % u(x,0)
 52% linearly decrease to zero to match with boundary conditions
 53u0(midPoint)       = u0m;
 54u0(1:midPoint-1)   = ((0:(midPoint-2))./midPoint)*u0m;
 55u0(midPoint+1:end) = (((midPoint-2):-1:0)./midPoint)*u0m;
 56 
 57
 58% Determine the boundary and initial functions
 59% du/dn(0,t) = q0j
 60q0j = zeros(N,1); % the flux 
 61%du/dn(1,t) = q1j
 62q1j = zeros(N,1); % the flux 
 63% u(0,t) = h0j
 64h0j =  zeros(N,1);
 65% u(1,t) = h1j
 66h1j =  zeros(N,1); 
 67elseif initCond==3
 68    E  = (sigIn)/2;      % int(u(x,t)dx) is a time function of size 1xN % assuming the temp drops to 0 at the boundaries linearly
 69% set f to be active only in the initial point 
 70f  = zeros(N,N0);    % function of N time points over N0 space points
 71f(:,1) = 1; % the source is only active at the mid point x=0.5;
 72% determine initial wave u(x,0) is the Phi function
 73
 74u0m      = max(sigIn(1:min(10,numel(sigIn)))); % take the max value 
 75u0       = u0m*exp(-linspace(0,1,N0));%ones(1,N0); % u(x,0)
 76
 77% Determine the boundary and initial functions
 78% du/dn(0,t) = q0j
 79q0j = zeros(N,1); % the flux in
 80%du/dn(1,t) = q1j
 81q1j = zeros(N,1); % the flux out
 82% u(0,t) = h0j
 83h0j = sigIn;%zeros(N,1);
 84% u(1,t) = h1j
 85h1j = zeros(N,1);
 86
 87else
 88    error('unsupported condition')
 89end
 90
 91% Discretize the time domain
 92T      = numel(sigIn);   % max time;
 93gridT  = (0:(N)).*(T/N); % N+1 boundary time points 
 94t      = zeros(1,N);     % the points on which the solution will be calculated
 95for tIdx = 1:N
 96    t(tIdx) = (gridT(tIdx)+gridT(tIdx+1))/2;
 97end
 98% Discretize the space domain
 99gridX = (0:(N0))./N0; % N0+1 space points
100x     = zeros(1,N0);  % the points on which the solution will be calculated
101for xIdx = 1:numel(x)
102    x(xIdx) = (gridX(xIdx+1)+gridX(xIdx))/2;
103end
104
105%===========================
106% Calculate the Boundary Elements
107
108% Calculate A and B, B*, and D
109[A,B,Bstar,D] = CalculateABBStarD(N,t,gridT,x,f);% call the mex
110
111% Calculate C
112C = CalculateC(N,N0,t,gridX);
113
114% Solve the equation
115% r =((X'*X+lambda*regTerm)^-1)*X'Y, with r the source and X and Y are to be
116% calculated
117
118% Calculate the components of X and Y
119[X,Y]= CalculateXY(N,N0,t,gridT,x,gridX,E,A,B,Bstar,C,D,u0,f,alpha);
120
121% get the regularization term 
122regTerm = GetRegularizationTerm(N,regOrder); % get RregTerm= 'R term
123
124% The source solution with regularization coefficient lambda
125r   = ((X')*X+lambda*regTerm)\(X'*Y);
126
127% Calculate the solution u(x,t)
128u  = CalculateHeatSolution(N0,N,t,gridT,x,u0,q0j,q1j,h0j,h1j,f,r);
129
130
131% plot source, signal and solution
132if all(plotFlag)
133  PlotResults(t,sigIn,u,r);
134end
135end
