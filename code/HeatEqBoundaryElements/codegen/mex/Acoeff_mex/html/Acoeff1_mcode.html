1function vals = Acoeff(ksi,j,x,t,timePoints)
 2% evaluate the coefficient A at the  boundary ksi=0 or ksi=1;
 3% for the index j which describes the time steps timePoints_j, at time t and space
 4% point x
 5% timePoints is a vector describing the time descritized domain
 6if t<=timePoints(j)
 7    vals=0;
 8elseif t>timePoints(j) && t<= timePoints(j+1) && x==ksi
 9    vals = sqrt((t-timePoints(j))/pi);
10elseif t>timePoints(j) && t<= timePoints(j+1) && x~=ksi
11    z0 = abs(x-ksi)/(2*sqrt(t-timePoints(j)));
12    vals = abs(x-ksi)/(2*sqrt(pi)) *(exp(-z0^2)/z0 - sqrt(pi)*(erfc(z0)));
13elseif t>timePoints(j+1) && x==ksi
14    vals = sqrt((t-timePoints(j))/pi)-sqrt((t-timePoints(j+1))/pi);
15elseif t>timePoints(j+1) && x~=ksi
16    z0 = abs(x-ksi)/(2*sqrt(t-timePoints(j)));
17    z1 = abs(x-ksi)/(2*sqrt(t-timePoints(j+1)));
18    vals = abs(x-ksi)/(2*sqrt(pi)) *(exp(-(z0^2))/z0 - exp(-(z1^2))/z1 +sqrt(pi)*(erf(z0)-erf(z1)));
19end
20end
